108 void exec() {
109 running = true;
110 while (running && (FCGX_Accept_r(&request) == 0)) {
111 char * method = FCGX_GetParam("REQUEST_METHOD", request.envp);
112 if (method == nullptr) {
113 fprintf(stderr,"Please add 'include fastcgi_params;' to the nginx conf.\n");
114 throw "JSONCGI parameters missing.\n";
115 }
116 if (strcmp(method, "GET") == 0) {
117 // create the header
118 std::string buffer = "Content-type: " + getCallback->getContentType();
119 buffer = buffer + "; charset=utf-8\r\n";
120 buffer = buffer + "\r\n";
121 // append the data
122 buffer = buffer + getCallback->getJSONString();
123 buffer = buffer + "\r\n";
124 // send the data to the web server
125 FCGX_PutStr(buffer.c_str(), buffer.length(), request.out);
126 FCGX_Finish_r(&request);
127 }
128 if (strcmp(method, "POST") == 0) {
129 long reqLen = 1;
130 char * content_length_str = FCGX_GetParam("CONTENT_LENGTH",
131 request.envp);

133 char* tmp = new char[reqLen];
134 FCGX_GetStr(tmp,reqLen,request.in);
135 tmp[reqLen - 1] = 0;
136 if (nullptr != postCallback) {
137 postCallback->postString(tmp);
138 }
139 delete[] tmp;
140 // create the header

142 buffer = buffer + "; charset=utf-8\r\n";
143 buffer = buffer + "\r\n";
144 // append the data
145 buffer = buffer + "\r\n";
146 buffer = buffer + "<html></html>\r\n";
147 // send the data to the web server
148 FCGX_PutStr(buffer.c_str(), buffer.length(), request.out);
